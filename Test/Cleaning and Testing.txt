---- Understanding and cleaning the data -----
--CUST_INFO

SELECT * FROM CUST_INFO;

-- Clean each column one by one
-- Check for duplicate IDs and NULL values
SELECT 
CST_ID,
COUNT(*)
FROM CUST_INFO
GROUP BY CST_ID
HAVING COUNT(*)>1;

SELECT * FROM CUST_INFO WHERE CST_ID = 29466;

SELECT *,
ROW_NUMBER()OVER(PARTITION BY CST_ID ORDER BY CST_CREATE_DATE DESC) AS FLAG
FROM CUST_INFO
WHERE CST_ID =29466;

SELECT * FROM
(
SELECT *,
ROW_NUMBER()OVER(PARTITION BY CST_ID ORDER BY CST_CREATE_DATE DESC) AS FLAG
FROM CUST_INFO
WHERE CST_ID IS NOT NULL
)AS t
WHERE FLAG =1;

-- Check for CST_KEY
--Expect 0 rows
SELECT 
CST_KEY
FROM CUST_INFO
WHERE TRIM(CST_KEY)!= CST_KEY;

-- Check for CST_FIRSTNAME
--Expect 0 rows
SELECT
CST_FIRSTNAME
FROM CUST_INFO
WHERE CST_FIRSTNAME != TRIM(CST_FIRSTNAME);


-- Check for CST_LASTNAME
--Expect 0 rows
SELECT
CST_LASTNAME
FROM CUST_INFO
WHERE TRIM(CST_LASTNAME)!=CST_LASTNAME;

-- Check for CST_MARITAL_STATUS

SELECT
DISTINCT CST_MARITAL_STATUS
FROM CUST_INFO;

SELECT
CASE 
WHEN TRIM(CST_MARITAL_STATUS) = 'M' THEN 'Married'
WHEN TRIM(CST_MARITAL_STATUS) = 'S' THEN 'Single'
ELSE 'NA'
END AS CST_MARITAL_STATUS
FROM CUST_INFO;

--Check for CST_GNDR

SELECT 
DISTINCT CST_GNDR
FROM CUST_INFO;

SELECT
CASE
WHEN TRIM(CST_GNDR) = 'M' THEN 'Male'
WHEN TRIM(CST_GNDR) = 'F' THEN 'Female'
ELSE 'NA'
END AS CST_GNDR
FROM CUST_INFO;

-- Finalized clean query for CUST_INFO
SELECT
CST_ID,
CST_KEY,
TRIM(CST_FIRSTNAME) AS CST_FIRSTNAME,
TRIM(CST_LASTNAME) AS CST_LASTNAME,
CASE 
WHEN TRIM(CST_MARITAL_STATUS) = 'M' THEN 'Married'
WHEN TRIM(CST_MARITAL_STATUS) = 'S' THEN 'Single'
ELSE 'NA'
END AS CST_MARITAL_STATUS,
CASE
WHEN TRIM(CST_GNDR) = 'M' THEN 'Male'
WHEN TRIM(CST_GNDR) = 'F' THEN 'Female'
ELSE 'NA'
END AS CST_GNDR,
CST_CREATE_DATE
FROM (
SELECT *,
ROW_NUMBER()OVER(PARTITION BY CST_ID ORDER BY CST_CREATE_DATE DESC) AS FLAG
FROM CUST_INFO
WHERE CST_ID IS NOT NULL
)AS t
WHERE FLAG =1;

-- Check for PRD_LINE

DISTINCT PRD_LINE
FROM PRD_INFO;

WHEN UPPER(TRIM(PRD_LINE)) = 'S' THEN 'Other Sales'
WHEN UPPER(TRIM(PRD_LINE)) = 'M' THEN 'Mountain'

END AS PRD_LINE

-- CHECK FOR PRD_START_DATE

SELECT * FROM PRD_INFO WHERE PRD_START_DT > PRD_END_DT;

SELECT * FROM PRD_INFO WHERE PRD_KEY = 'AC-HE-HL-U509-R';

SELECT *,
LEAD(PRD_START_DT)OVER(PARTITION BY PRD_KEY ORDER BY PRD_START_DT)-1 AS PRD_END_DT_TEST
FROM PRD_INFO
WHERE PRD_KEY = 'AC-HE-HL-U509-R';

-- Finalized clean query for PRD_INFO

PRD_ID,
REPLACE(SUBSTR(PRD_KEY,1,5),'-','_') AS PRD_KEY,
SUBSTR(PRD_KEY,7,LENGTH(PRD_KEY))AS CAT_ID,

WHEN UPPER(TRIM(PRD_LINE)) = 'M' THEN 'Mountain'
WHEN UPPER(TRIM(PRD_LINE)) = 'T' THEN 'Touring'

PRD_START_DT,
LEAD(PRD_START_DT)OVER(PARTITION BY PRD_KEY ORDER BY PRD_START_DT)-1 AS PRD_END_DT

--Let's clean SALES_DETAILS

SELECT * FROM SALES_DETAILS;

-- Check for SLS_ORD_NUM

SELECT
SLS_ORD_NUM
FROM SALES_DETAILS
WHERE SLS_ORD_NUM != TRIM(SLS_ORD_NUM);

-- Check for SLS_PRD_KEY

SELECT
SLS_PRD_KEY
FROM SALES_DETAILS
WHERE SLS_PRD_KEY != TRIM(SLS_PRD_KEY);

-- Check for SLS_CUST_ID

SELECT
SLS_CUST_ID
FROM SALES_DETAILS
WHERE SLS_CUST_ID IS NULL;

-- Check for SLS_ORDER_DT

SELECT
SLS_ORDER_DT
FROM SALES_DETAILS
WHERE SLS_ORDER_DT IS NULL OR LENGTH(SLS_ORDER_DT)!=8 OR SLS_ORDER_DT<=0 OR SLS_ORDER_DT >20500101 OR SLS_ORDER_DT <19000101;

SELECT SLS_ORDER_DT FROM SALES_DETAILS WHERE SLS_ORDER_DT <=0 OR LEN(SLS_ORDER_DT) <8;

SELECT DISTINCT
CASE
WHEN SLS_ORDER_DT <= 0 OR LENGTH(SLS_ORDER_DT)!=8 THEN NULL
ELSE TO_DATE(SLS_ORDER_DT::VARCHAR,'YYYYMMDD')
END AS SLS_ORDER_DT
FROM SALES_DETAILS;

-- Check for SLS_SHIP_DT

SELECT
SLS_SHIP_DT
FROM SALES_DETAILS
WHERE SLS_SHIP_DT IS NULL OR SLS_SHIP_DT <=0 OR LENGTH(SLS_SHIP_DT)!=8 OR SLS_SHIP_DT >20500101 OR SLS_SHIP_DT <19000101;

-- Check for SLS_DUE_DT

SELECT
SLS_DUE_DT
FROM SALES_DETAILS
WHERE SLS_DUE_DT IS NULL OR SLS_DUE_DT <=0 OR LENGTH(SLS_DUE_DT)!=8 OR SLS_DUE_DT >20500101 OR SLS_DUE_DT <19000101;

-- Check for invalid dates

SELECT * FROM SALES_DETAILS
WHERE SLS_ORDER_DT > SLS_DUE_DT OR SLS_ORDER_DT > SLS_SHIP_DT;

-- Check for SLS_SALES
-- SALES must be equal to quantity*price 
-- No negative or null values are allowed

SELECT 
SLS_SALES,
SLS_PRICE,
SLS_QUANTITY
FROM SALES_DETAILS
WHERE SLS_SALES != SLS_QUANTITY* SLS_PRICE OR SLS_SALES IS NULL OR SLS_SALES<=0 OR SLS_QUANTITY IS NULL OR SLS_QUANTITY <=0 OR SLS_PRICE IS NULL OR SLS_PRICE <=0;


SELECT
CASE 
WHEN SLS_SALES IS NULL OR SLS_SALES <=0 OR SLS_SALES != SLS_QUANTITY* ABS(SLS_PRICE)
THEN SLS_QUANTITY* ABS(SLS_PRICE)
ELSE SLS_SALES
END AS SLS_SALES,
SLS_QUANTITY,
CASE
WHEN SLS_PRICE IS NULL OR SLS_PRICE <=0 OR SLS_PRICE != SLS_SALES/SLS_QUANTITY
THEN SLS_SALES/NULLIFZERO(SLS_QUANTITY)
ELSE ABS(SLS_PRICE)
END AS SLS_PRICE
FROM SALES_DETAILS;

-- Finalized clean query for SALES_DETAILS

SELECT
SLS_ORD_NUM,
SLS_PRD_KEY,
SLS_CUST_ID,
CASE
WHEN SLS_ORDER_DT = 0 OR LENGTH(SLS_ORDER_DT)!=8 THEN NULL
ELSE TO_DATE(SLS_ORDER_DT::VARCHAR,'YYYYMMDD')  
END AS SLS_ORDER_DT,
CASE
WHEN SLS_SHIP_DT = 0 OR LENGTH(SLS_SHIP_DT)!=8 THEN NULL
ELSE TO_DATE(SLS_SHIP_DT::VARCHAR,'YYYYMMDD')  
END AS SLS_SHIP_DT,
CASE
WHEN SLS_DUE_DT = 0 OR LENGTH(SLS_DUE_DT)!=8 THEN NULL
ELSE TO_DATE(SLS_DUE_DT::VARCHAR,'YYYYMMDD')  
END AS SLS_DUE_DT,
CASE 
WHEN SLS_SALES IS NULL OR SLS_SALES <=0 OR SLS_SALES != SLS_QUANTITY* ABS(SLS_PRICE)
THEN SLS_QUANTITY* ABS(SLS_PRICE)
ELSE SLS_SALES
END AS SLS_SALES,
SLS_QUANTITY,
CASE
WHEN SLS_PRICE IS NULL OR SLS_PRICE <=0 OR SLS_PRICE != SLS_SALES/SLS_QUANTITY
THEN ROUND(SLS_SALES/NULLIFZERO(SLS_QUANTITY),0)
ELSE ROUND(ABS(SLS_PRICE),0)
END AS SLS_PRICE
FROM SALES_DETAILS;
